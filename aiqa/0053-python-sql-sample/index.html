    PythonсЂ«sqlite3ТеЎТ║ќсЃЕсѓцсЃќсЃЕсЃфсѓњСй┐сЂБсЂдсђЂсЂЊсѓїсѓЅсЂ«ТЊЇСйюсѓњсЂЙсЂесѓЂсЂЪсѓхсЃ│сЃЌсЃФсѓ│сЃ╝сЃЅсѓњСйюТѕљсЂЌсЂЙсЂЌсЂЪсђѓ

    SQLiteсЂ»сђЂТјЦуХџ№╝ѕconnect№╝ЅсЂЌсЂЪтЙїсЂФсђЂтЉйС╗цсѓњт«ЪУАїсЂЎсѓІсЂЪсѓЂсЂ«сѓФсЃ╝сѓйсЃФ№╝ѕcursor№╝ЅсѓњтЈќтЙЌсЂЌсЂдТЊЇСйюсЂЎсѓІсЂ«сЂїтЪ║ТюгсЂ«ТхЂсѓїсЂДсЂЎсђѓ

    SQLiteТЊЇСйюсЂ«Pythonсѓ│сЃ╝сЃЅСЙІ

    import sqlite3

def get_connection():
    # 'my_database.db' сЂесЂёсЂєсЃЋсѓАсѓцсЃФтљЇсЂДDBсѓњСйюТѕљ/ТјЦуХџ
    return sqlite3.connect('my_database.db')

def create_table():
    conn = get_connection()
    cur = conn.cursor()
    cur.execute("""
    CREATE TABLE IF NOT EXISTS posts (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        text TEXT NOT NULL,
        created_at DATETIME DEFAULT (DATETIME('now', 'localtime'))
    )
    """)
    conn.commit()
    conn.close()

# 1. Тќ░сЂЌсЂё text сѓњУ┐йтіасЂЎсѓІ
def add_post(text):
    conn = get_connection()
    cur = conn.cursor()
    cur.execute("INSERT INTO posts (text) VALUES (?)", (text,))
    conn.commit()
    conn.close()

# 2. сЂЎсЂ╣сЂдсЂ«сЃгсѓ│сЃ╝сЃЅсѓњтЈќтЙЌсЂЎсѓІ
def get_all_posts():
    conn = get_connection()
    cur = conn.cursor()
    cur.execute("SELECT * FROM posts ORDER BY created_at DESC")
    rows = cur.fetchall()
    conn.close()
    return rows

# 3. textсѓњсѓГсЃ╝сЃ»сЃ╝сЃЅсЂДТцюу┤бсЂЎсѓІ
def search_posts(keyword):
    conn = get_connection()
    cur = conn.cursor()
    # жЃетѕєСИђУЄ┤сЂ«сЂЪсѓЂсђЂсѓГсЃ╝сЃ»сЃ╝сЃЅсѓњ % сЂДтЏ▓сѓђ
    query_param = f"%{keyword}%"
    cur.execute("SELECT * FROM posts WHERE text LIKE (?) ORDER BY created_at DESC", (query_param,))
    rows = cur.fetchall()
    conn.close()
    return rows

# 4. сЃгсѓ│сЃ╝сЃЅсѓњтЅіжЎцсЂЎсѓІ
def delete_post(post_id):
    conn = get_connection()
    cur = conn.cursor()
    cur.execute("DELETE FROM posts WHERE id = (?)", (post_id,))
    conn.commit()
    conn.close()

# --- тІЋСйюуб║УфЇ ---
if __name__ == "__main__":
    create_table()

    # У┐йтіа
    add_post("сЂЊсѓЊсЂФсЂАсЂ»сђЂPythonсЂ«СИќуЋї№╝Ђ")
    add_post("SQLiteсЂ»сЂесЂдсѓѓСЙ┐тѕЕсЂДсЂЎсЂГсђѓ")

    # Тцюу┤б№╝ѕсђїPythonсђЇсѓњтљФсѓђсѓѓсЂ«№╝Ѕ
    print("--- Тцюу┤бухљТъю ---")
    results = search_posts("Python")
    for row in results:
        print(row)

    # тЁеС╗ХтЈќтЙЌ
    print("\n--- тЁеС╗ХУАеуц║ ---")
    all_posts = get_all_posts()
    for row in all_posts:
        print(row)

    # тЅіжЎц№╝ѕСЙІсЂесЂЌсЂдТюђтѕЮсЂ«1С╗ХсѓњтЅіжЎц№╝Ѕ
    if all_posts:
        delete_post(all_posts[0][0])
        print(f"\nID {all_posts[0][0]} сѓњтЅіжЎцсЂЌсЂЙсЂЌсЂЪсђѓ")
